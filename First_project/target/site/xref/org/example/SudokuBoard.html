3   public class SudokuBoard {
4       //private final int size = 9;
5       private final int[][] sudokuBoard = new int [9][9];
6       private SudokuSolver solver;
7   
8   
9   
10      public SudokuBoard() {
11      }
12  
13      public int getCell(int x, int y) {
14          if (x >= 0 && x < 9 && y >= 0 && y < 9) {
15              return sudokuBoard[x][y];
16          } else {
17              return 0;
18          }
19      }
20  
21      public void setCell(int x, int y, int value) throws IndexOutRange {
22          if (x >= 0 && x < 9 && y >= 0 && y < 9 && value >= 1 && value <= 9) {
23              sudokuBoard[x][y] = value;
24          } else if (x < 0 || x >= 9) {
25              throw new IndexOutRange("Wrong column number:" + x);
26          } else if (y < 0 || y >= 9) {
27              throw new IndexOutRange("Wrong column number:" + y);
28          } else {
29              sudokuBoard[x][y] = 0;
30          }
31      }
32  
33      public void solveGame() throws IndexOutRange {
34          this.solver = new BackTrackingSudokuSolver();
35          solver.solve(this);
36      }
37  
38      public boolean checkCell(int num, int x, int y) {
39          for (int i = 0; i < 9; i++) {
40              if (getCell(i, y) == num) {
41                  return false;
42              }
43          }
44          for (int i = 0; i < 9; i++) {
45              if (getCell(x, i) == num) {
46                  return false;
47              }
48          }
49          for (int i = 0; i <= 2; i++) {
50              for (int j = 0; j <= 2; j++) {
51                  if (getCell(i + x - x % 3, j + y - y % 3) == num) {
52                      return false;
53                  }
54              }
55          }
56          return true;
57      }
58  }
